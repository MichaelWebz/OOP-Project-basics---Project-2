
#include <iostream>
#include <string>
#include <vector>
using namespace std;

struct Employee {
    int id;
    string name;
    string position;
    double salary;
};

vector<Employee> employees;

void addEmployee() {
    Employee emp;
    cout << "\nEnter Employee ID: ";
    cin >> emp.id;
    cin.ignore(); // لتفادي مشاكل الـ getline بعد cin
    cout << "Enter Name: ";
    getline(cin, emp.name);
    cout << "Enter Position: ";
    getline(cin, emp.position);
    cout << "Enter Salary: ";
    cin >> emp.salary;

    employees.push_back(emp);
    cout << "✅ Employee added successfully!\n";
}

void listEmployees() {
    if (employees.empty()) {
        cout << "\n⚠ No employees found.\n";
        return;
    }
    cout << "\n--- Employee List ---\n";
    for (auto &emp : employees) {
        cout << "ID: " << emp.id
             << ", Name: " << emp.name
             << ", Position: " << emp.position
             << ", Salary: " << emp.salary << endl;
    }
}

void searchEmployee() {
    int id;
    cout << "\nEnter Employee ID to search: ";
    cin >> id;

    for (auto &emp : employees) {
        if (emp.id == id) {
            cout << "Found: " << emp.name << " | " << emp.position 
                 << " | Salary: " << emp.salary << endl;
            return;
        }
    }
    cout << "❌ Employee not found.\n";
}

void editEmployee() {
    int id;
    cout << "\nEnter Employee ID to edit: ";
    cin >> id;

    for (auto &emp : employees) {
        if (emp.id == id) {
            cin.ignore();
            cout << "Enter New Name: ";
            getline(cin, emp.name);
            cout << "Enter New Position: ";
            getline(cin, emp.position);
            cout << "Enter New Salary: ";
            cin >> emp.salary;
            cout << "✅ Employee updated successfully!\n";
            return;
        }
    }
    cout << "❌ Employee not found.\n";
}

void adminMenu() {
    int choice;
    do {
        cout << "\n=== Admin Menu ===\n";
        cout << "1. Add Employee\n";
        cout << "2. Search Employee\n";
        cout << "3. Edit Employee\n";
        cout << "4. List All Employees\n";
        cout << "0. Exit\n";
        cout << "Choose: ";
        cin >> choice;

        switch (choice) {
            case 1: addEmployee(); break;
            case 2: searchEmployee(); break;
            case 3: editEmployee(); break;
            case 4: listEmployees(); break;
            case 0: cout << "Exiting...\n"; break;
            default: cout << "⚠ Invalid choice.\n";
        }
    } while (choice != 0);
}

int main() {
    // هنا ممكن تعمل تسجيل دخول للـ Admin لو عايز
    adminMenu();
    return 0;
}
